ALTER FUNCTION [dbo].[GetPriceSession] 
(
	@StudentID int,
	@AID int
)
RETURNS int 
AS
BEGIN
	DECLARE @Price as Int ; 
	Declare @GroupID int = (Select  GroupID  from Attendance Where ID = @AID);
	Declare @Date nvarchar(100)  = (Select Date from Attendance where ID = @AID);
	Declare @ClassID int = (Select  ClassID  from Groups Where GroupID = @GroupID);
	
	 IF Exists(Select CPrice From Discounts Where StudentID = @StudentID
											 And ClassID = @ClassID 
											 and CONVERT(DATE,StudentDate, 105) <= CONVERT(DATE,@Date, 105)
											 And (Done is null or CONVERT(DATE,Done, 105) > CONVERT(DATE,@Date, 105) ))
	 Begin 
			Set @Price  =(Select CPrice/4 From Discounts Where StudentID = @StudentID
											 And ClassID = @ClassID 
											 and CONVERT(DATE,StudentDate, 105) <= CONVERT(DATE,@Date, 105)
											 And (Done is null or  CONVERT(DATE,Done, 105) > CONVERT(DATE,@Date, 105) ))
	 End
     else 
     Begin
	 		Set @Price  =(Select CPrice/4 From Class Where ID = @ClassID)
	 End;
	RETURN @Price

END
